function genWorld(){
  const map=[];for(let x=0;x<WW;x++)map[x]=new Array(WH).fill(0);
  const surf=[];for(let x=0;x<WW;x++)surf[x]=Math.round(75+Math.sin(x*.028)*10+Math.sin(x*.07+1.1)*5+Math.sin(x*.18+2.3)*3+Math.sin(x*.4+.7)*1.5);
  for(let x=0;x<WW;x++)for(let y=0;y<WH;y++){if(y>=WH-1){map[x][y]=10;continue;}if(y>surf[x]+25)map[x][y]=3;else if(y>surf[x])map[x][y]=2;else if(y===surf[x])map[x][y]=1;}

  for(let i=0;i<380;i++){const x=(1+Math.random()*(WW-2))|0,y=(surf[x]+10+Math.random()*(WH-surf[x]-15))|0;const ore=Math.random()<.6?6:7;for(let dx=-1;dx<=1;dx++)for(let dy=-1;dy<=1;dy++){const nx=x+dx,ny=y+dy;if(nx>0&&nx<WW&&ny>0&&ny<WH&&map[nx][ny]===3&&Math.random()<.7)map[nx][ny]=ore;}}
  for(let x=5;x<WW-5;x++){if(Math.random()<.08&&map[x][surf[x]]===1&&Math.abs(surf[x]-surf[x-1])<=2&&Math.abs(surf[x]-surf[x+1])<=2){const h=5+((Math.random()*4)|0);for(let y=surf[x]-h;y<surf[x];y++)if(y>=0)map[x][y]=4;const lr=3+(Math.random()*2|0),lt=surf[x]-h-1;for(let dx=-lr;dx<=lr;dx++)for(let dy=-lr;dy<=lr;dy++){if(dx*dx*.6+dy*dy>lr*lr+1)continue;const lx=x+dx,ly=lt+dy;if(lx>0&&lx<WW&&ly>0&&ly<WH&&map[lx][ly]===0)map[lx][ly]=5;}}}

class Mob{
  constructor(scene,x,y,type){
    this.sc=scene;this.type=type;this.alive=true;this.dead=false;
    this.cont=scene.add.container(x,y).setDepth(11);this.gfx=scene.add.graphics();this.cont.add(this.gfx);this.hpGfx=scene.add.graphics();this.cont.add(this.hpGfx);
    this.vx=0;this.vy=0;this.onG=false;this.facing=1;this.stateT=0;this.state='patrol';this.dmgCD=0;this.invCD=0;
    switch(type){case'goblin':this.hp=this.maxHp=35;this.color=0x228822;this.w=8;this.h=20;this.dmg=8;break;case'slime':this.hp=this.maxHp=20;this.color=0x3355cc;this.w=10;this.h=10;this.dmg=5;break;case'zombie':this.hp=this.maxHp=60;this.color=0x446622;this.w=8;this.h=22;this.dmg=12;break;case'eye':this.hp=this.maxHp=45;this.color=0xcc2222;this.w=12;this.h=12;this.dmg=10;this.flying=true;break;default:this.hp=this.maxHp=20;this.color=0x888888;this.w=8;this.h=16;this.dmg=6;}
    this.draw();
  }
  solid(tx,ty){if(tx<0||tx>=WW||ty<0||ty>=WH)return true;const t=this.sc.map[tx][ty];return t!==0&&t!==9&&t!==5&&t!==11;}
  draw(){
    const g=this.gfx;g.clear();if(!this.alive)return;
    if(this.type==='goblin'){const ls=this.state==='chase'?Math.sin(Date.now()*.018)*14:0;g.fillStyle(0x000000,.15);g.fillEllipse(0,11,12,4);g.fillStyle(0x114411);g.fillRect(-5,8,4,6+ls*.3);g.fillRect(1,8,4,6-ls*.3);g.fillStyle(this.color);g.fillRect(-4,-8,8,14);g.fillStyle(0x33aa33);g.fillRect(-4,-8,8,3);g.fillStyle(0x003300);g.fillRect(-5,-8,4,8);g.fillRect(1,-8,4,8);g.fillStyle(0xddff88);g.fillRect(-3,-12,6,6);g.fillStyle(0x113311);g.fillRect(-3,-12,6,2);g.fillStyle(0xffdd00);g.fillRect(0,1,5,2);g.scaleX=this.facing;}
    else if(this.type==='slime'){const bob=Math.sin(Date.now()*.006)*(this.state==='chase'?3:1.5);g.fillStyle(0x001a44,.2);g.fillEllipse(0,8,14,4);g.fillStyle(0x2244aa);g.fillEllipse(0,bob,14,10);g.fillStyle(0x3366ff,.8);g.fillEllipse(0,bob-2,11,7);g.fillStyle(0x66aaff,.4);g.fillEllipse(-1,bob-3,5,3);g.fillStyle(0xffff88);g.fillCircle(-2,bob-1,2);g.fillCircle(2,bob-1,2);g.fillStyle(0x000000);g.fillCircle(-2,bob-1,1);g.fillCircle(2,bob-1,1);}
    else if(this.type==='zombie'){const ls=this.state==='chase'?Math.sin(Date.now()*.012)*10:0;g.fillStyle(0x000000,.2);g.fillEllipse(0,12,14,4);g.fillStyle(0x223310);g.fillRect(-4,8,4,8+ls*.3);g.fillRect(1,8,4,8-ls*.3);g.fillStyle(0x334422);g.fillRect(-5,-4,10,13);g.fillStyle(0x446633);g.fillRect(-6,-5,4,10+ls*.2);g.fillRect(2,-5,4,10-ls*.2);g.fillStyle(0x88aa66);g.fillRect(-4,-16,8,13);g.fillStyle(0xcc2200,.8);g.fillCircle(-1,-11,2);g.fillCircle(2,-11,2);g.scaleX=this.facing;}
    else if(this.type==='eye'){const fl=Math.sin(Date.now()*.004)*4;g.fillStyle(0x000000,.2);g.fillEllipse(0,10,16,4);g.fillStyle(0xaa1111);g.fillEllipse(0,fl,16,14);g.fillStyle(0xdd2222);g.fillEllipse(0,fl-1,12,10);g.fillStyle(0xffffff);g.fillEllipse(0,fl-1,8,7);g.fillStyle(0x000000);g.fillEllipse(0,fl,5,5);g.fillStyle(0xff4444,.6);g.fillEllipse(0,fl+2,3,3);}
    const hg=this.hpGfx;hg.clear();if(this.hp<this.maxHp){const fy=-this.h-8;hg.fillStyle(0x660000);hg.fillRect(-12,fy,24,4);hg.fillStyle(0x44ff44);hg.fillRect(-12,fy,24*(this.hp/this.maxHp),4);}
  }
  update(dt,px,py){
    if(!this.alive)return;if(this.dmgCD>0)this.dmgCD-=dt;if(this.invCD>0)this.invCD-=dt;this.stateT+=dt;
    const dx=px-this.cont.x,dy=py-this.cont.y,dist=Math.sqrt(dx*dx+dy*dy);
    if(dist<110)this.state='chase';else if(dist>220)this.state='patrol';
    const spd=this.state==='chase'?({goblin:2,slime:.9,zombie:1.2,eye:1.5}[this.type]||1.5):.6;
    if(this.type==='eye'){if(this.state==='chase'){const ang=Math.atan2(dy,dx);this.cont.x+=Math.cos(ang)*spd;this.cont.y+=Math.sin(ang)*spd;}else{this.cont.x+=this.facing*spd*.4;if(this.stateT>1500){this.facing*=-1;this.stateT=0;}}this.draw();return;}
    if(this.state==='chase'){this.facing=dx>0?1:-1;this.vx=this.facing*spd;}else{if(this.stateT>1800){this.facing*=-1;this.stateT=0;}this.vx=this.facing*spd*.5;}
    this.vy=Math.min(this.vy+.35,16);
    const nnx=this.cont.x+this.vx,bx=Math.floor(nnx/TS),by=Math.floor((this.cont.y-this.h*.5)/TS),by2=Math.floor(this.cont.y/TS);
    if(!this.solid(bx,by)&&!this.solid(bx,by2))this.cont.x=Phaser.Math.Clamp(nnx,0,WW*TS);else{this.vx=0;this.facing*=-1;}
    if(this.onG&&this.solid(Math.floor((this.cont.x+this.facing*TS*1.2)/TS),Math.floor(this.cont.y/TS)))this.vy=-6.5;
    const nny=this.cont.y+this.vy;this.onG=false;
    if(!this.solid(Math.floor(this.cont.x/TS),Math.floor(nny/TS)))this.cont.y=nny;else{if(this.vy>0)this.onG=true;this.vy=0;}
    this.draw();
  }
  takeDamage(dmg,vx,vy){
    if(!this.alive||this.invCD>0)return;this.hp-=dmg;this.invCD=200;
    if(this.hp<=0){this.alive=false;this.sc.time.delayedCall(350,()=>{this.gfx.clear();this.dead=true;});this.sc.time.delayedCall(30,()=>this.sc.breakFX(this.cont.x,this.cont.y,{a:this.color,b:0xcc2222}));const loot={goblin:['coal','iron'],slime:['dirt'],zombie:['plank','iron'],eye:['iron','coal']}[this.type]||['dirt'];this.sc.spawnDrop(this.cont.x,this.cont.y,loot[Math.floor(Math.random()*loot.length)]);if(Math.random()<.2)this.sc.spawnDrop(this.cont.x,this.cont.y,'bandage');}else{this.vx=vx||0;this.vy=vy||-3;}
  }
}
